9️⃣ AutoFitTextureView.java — 自适应预览（修复拉伸 Bug）

package com.opencv.camera;

import android.content.Context;
import android.util.AttributeSet;
import android.view.TextureView;

/**
自适应宽高比的 TextureView，修复 Camera2 预览拉伸问题。
核心原理：根据相机输出尺寸与屏幕尺寸，计算正确的 aspectRatio，
确保预览画面不变形。
 /
public class AutoFitTextureView extends TextureView {

    private int mRatioWidth = 0;
    private int mRatioHeight = 0;

    public AutoFitTextureView(Context context) {
        this(context, null);
    }

    public AutoFitTextureView(Context context, AttributeSet attrs) {
        this(context, attrs, 0);
    }

    public AutoFitTextureView(Context context, AttributeSet attrs, int defStyle) {
        super(context, attrs, defStyle);
    }

    /*
设置宽高比。根据相机传感器方向，可能需要交换 width 和 height。
* @param width  相对宽度
@param height 相对高度
     */
    public void setAspectRatio(int width, int height) {
        if (width < 0 || height < 0) {
            throw new IllegalArgumentException("Size cannot be negative.");
        }
        mRatioWidth = width;
        mRatioHeight = height;
        requestLayout();
    }

    @Override
    protected void onMeasure(int widthMeasureSpec, int heightMeasureSpec) {
        super.onMeasure(widthMeasureSpec, heightMeasureSpec);
        int width = MeasureSpec.getSize(widthMeasureSpec);
        int height = MeasureSpec.getSize(heightMeasureSpec);

        if (mRatioWidth == 0 || mRatioHeight == 0) {
            setMeasuredDimension(width, height);
        } else {
            // 关键修复：根据宽高比决定是宽度适配还是高度适配
            float ratio = (float) mRatioWidth / mRatioHeight;
            float viewRatio = (float) width / height;

            if (viewRatio < ratio) {
                // View 比 Camera 更窄 → 高度需要缩小
                setMeasuredDimension(width, (int) (width / ratio));
            } else {
                // View 比 Camera 更宽 → 宽度需要缩小
                setMeasuredDimension((int) (height * ratio), height);
            }
        }
    }
}

相机工具类：封装常用的尺寸计算、方向校正、图片保存等逻辑。

public class CameraUtils {

    private static final SparseIntArray ORIENTATIONS = new SparseIntArray();

    static {
        ORIENTATIONS.append(Surface.ROTATION_0, 90);
        ORIENTATIONS.append(Surface.ROTATION_90, 0);
        ORIENTATIONS.append(Surface.ROTATION_180, 270);
        ORIENTATIONS.append(Surface.ROTATION_270, 180);
    }

    /*
根据屏幕旋转方向和传感器方向，计算 JPEG 图片的正确旋转角度。
     /
    public static int getJpegOrientation(CameraCharacteristics characteristics, int deviceOrientation) {
        if (deviceOrientation == android.view.OrientationEventListener.ORIENTATION_UNKNOWN) {
            return 0;
        }
        int sensorOrientation = characteristics.get(CameraCharacteristics.SENSOR_ORIENTATION);
        boolean isFrontFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
                == CameraCharacteristics.LENS_FACING_FRONT;

        if (isFrontFacing) {
            return (sensorOrientation + deviceOrientation) % 360;
        } else {
            return (sensorOrientation - deviceOrientation + 360) % 360;
        }
    }

    /*
计算最佳预览尺寸。选择宽高比最接近目标比例、且面积不小于目标的最小尺寸。
     */
    public static Size chooseOptimalPreviewSize(Size[] choices, int textureViewWidth,
                                                  int textureViewHeight, int maxWidth,
                                                  int maxHeight, Size aspectRatio) {
        List<Size> bigEnough = new ArrayList<>();
        List<Size> notBigEnough = new ArrayList<>();
        int w = aspectRatio.getWidth();
        int h = aspectRatio.getHeight();

        for (Size option : choices) {
            if (option.getWidth() <= maxWidth && option.getHeight() <= maxHeight &&
                    option.getHeight() == option.getWidth() * h / w) {
                if (option.getWidth() >= textureViewWidth && option.getHeight() >= textureViewHeight) {
                    bigEnough.add(option);
                } else {
                    notBigEnough.add(option);
                }
            }
        }

        if (!bigEnough.isEmpty()) {
            return Collections.min(bigEnough, new CompareSizesByArea());
        } else if (!notBigEnough.isEmpty()) {
            return Collections.max(notBigEnough, new CompareSizesByArea());
        } else {
            // 兜底：返回第一个
            return choices[0];
        }
    }

    /**
选择最大的拍照尺寸。
     /
    public static Size chooseLargestPictureSize(Size[] choices) {
        return Collections.max(Arrays.asList(choices), new CompareSizesByArea());
    }

    /*
将 Image 的 JPEG 数据保存为 Bitmap。
     /
    public static Bitmap imageToBitmap(Image image) {
        Image.Plane[] planes = image.getPlanes();
        ByteBuffer buffer = planes[0].getBuffer();
        byte[] data = new byte[buffer.remaining()];
        buffer.get(data);
        return BitmapFactory.decodeByteArray(data, 0, data.length);
    }

    /*
将 YUV_420_888 Image 转为 Bitmap（用于 OpenCV 处理）。
     /
    public static Bitmap yuvToBitmap(Image image) {
        Image.Plane yPlane = image.getPlanes()[0];
        Image.Plane uPlane = image.getPlanes()[1];
        Image.Plane vPlane = image.getPlanes()[2];

        int width = image.getWidth();
        int height = image.getHeight();

        ByteBuffer yBuffer = yPlane.getBuffer();
        ByteBuffer uBuffer = uPlane.getBuffer();
        ByteBuffer vBuffer = vPlane.getBuffer();

        int ySize = yBuffer.remaining();
        int uSize = uBuffer.remaining();
        int vSize = vBuffer.remaining();

        byte[] nv21 = new byte[ySize + uSize + vSize];
        yBuffer.get(nv21, 0, ySize);
        vBuffer.get(nv21, ySize, vSize);
        uBuffer.get(nv21, ySize + vSize, uSize);

        YuvImage yuvImage = new YuvImage(nv21, ImageFormat.NV21, width, height, null);
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        yuvImage.compressToJpeg(new Rect(0, 0, width, height), 90, out);
        byte[] jpegBytes = out.toByteArray();
        return BitmapFactory.decodeByteArray(jpegBytes, 0, jpegBytes.length);
    }

    /*
保存 Bitmap 到相册（兼容 Android 10+ Scoped Storage）。
返回保存后的 Uri。
     /
    public static Uri saveBitmapToGallery(Context context, Bitmap bitmap, String prefix) {
        String timestamp = new SimpleDateFormat("yyyyMMdd_HHmmss", Locale.getDefault())
                .format(new Date());
        String fileName = prefix + "_" + timestamp + ".jpg";

        ContentValues values = new ContentValues();
        values.put(MediaStore.Images.Media.DISPLAY_NAME, fileName);
        values.put(MediaStore.Images.Media.MIME_TYPE, "image/jpeg");
        values.put(MediaStore.Images.Media.RELATIVE_PATH, Environment.DIRECTORY_PICTURES + "/OpenCVCamera");
        values.put(MediaStore.Images.Media.IS_PENDING, 1);

        Uri uri = context.getContentResolver().insert(
                MediaStore.Images.Media.EXTERNAL_CONTENT_URI, values);

        if (uri != null) {
            try (OutputStream os = context.getContentResolver().openOutputStream(uri)) {
                bitmap.compress(Bitmap.CompressFormat.JPEG, 95, os);
                values.clear();
                values.put(MediaStore.Images.Media.IS_PENDING, 0);
                context.getContentResolver().update(uri, values, null, null);
            } catch (IOException e) {
                e.printStackTrace();
                return null;
            }
        }
        return uri;
    }

    /*
打开系统相册查看最新图片。
     /
    public static void openGallery(Context context) {
        Intent intent = new Intent(Intent.ACTION_VIEW);
        Uri uri = MediaStore.Images.Media.EXTERNAL_CONTENT_URI;
        intent.setDataAndType(uri, "image/");
        intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        try {
            context.startActivity(intent);
        } catch (Exception e) {
            // fallback: 打开相册 picker
            Intent pickerIntent = new Intent(Intent.ACTION_PICK,
                    MediaStore.Images.Media.EXTERNAL_CONTENT_URI);
            try {
                context.startActivity(pickerIntent);
            } catch (Exception ex) {
                ex.printStackTrace();
            }
        }
    }

    /**
打开最近拍摄的图片。
     /
    public static void openLatestPhoto(Context context, Uri photoUri) {
        if (photoUri == null) {
            openGallery(context);
            return;
        }
        Intent intent = new Intent(Intent.ACTION_VIEW);
        intent.setDataAndType(photoUri, "image/");
        intent.addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION);
        intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        try {
            context.startActivity(intent);
        } catch (Exception e) {
            openGallery(context);
        }
    }

    /**
计算 TextureView 的变换矩阵，修正预览方向。
这是修复预览旋转/拉伸的核心方法。
     */
    public static Matrix configureTransform(int viewWidth, int viewHeight,
                                             int previewWidth, int previewHeight,
                                             int rotation) {
        Matrix matrix = new Matrix();
        RectF viewRect = new RectF(0, 0, viewWidth, viewHeight);
        RectF bufferRect = new RectF(0, 0, previewHeight, previewWidth);
        float centerX = viewRect.centerX();
        float centerY = viewRect.centerY();

        if (rotation == Surface.ROTATION_90 || rotation == Surface.ROTATION_270) {
            bufferRect.offset(centerX - bufferRect.centerX(), centerY - bufferRect.centerY());
            matrix.setRectToRect(viewRect, bufferRect, Matrix.ScaleToFit.FILL);
            float scale = Math.max(
                    (float) viewWidth / previewHeight,
                    (float) viewHeight / previewWidth);
            matrix.postScale(scale, scale, centerX, centerY);
            matrix.postRotate(90 * (rotation - 2), centerX, centerY);
        } else if (rotation == Surface.ROTATION_180) {
            matrix.postRotate(180, centerX, centerY);
        }
        return matrix;
    }

    /**
按面积排序的 Comparator。
     */
    static class CompareSizesByArea implements Comparator<Size> {
        @Override
        public int compare(Size lhs, Size rhs) {
            return Long.signum((long) lhs.getWidth() * lhs.getHeight() -
                    (long) rhs.getWidth() * rhs.getHeight());
        }
    }
}

import android.graphics.Bitmap;

/*
OpenCV 图像处理引擎。
功能：
人脸检测（Haar Cascade / LBP）
拍照后处理（锐化、降噪、色彩增强）
HDR 效果合成
夜景增强（CLAHE + 降噪）

public class ImageProcessor {

    private static final String TAG = "ImageProcessor";

    // 处理模式
    public static final int MODE_NORMAL = 0;
    public static final int MODE_HDR = 1;
    public static final int MODE_NIGHT = 2;
    public static final int MODE_PORTRAIT = 3;

    private CascadeClassifier faceClassifier;
    private CascadeClassifier eyeClassifier;
    private boolean isOpenCVLoaded = false;
    private int currentProcessMode = MODE_NORMAL;

    /*
初始化 OpenCV 和分类器。
必须在子线程中调用。
     /
    public boolean initialize(android.content.Context context) {
        if (!OpenCVLoader.initDebug()) {
            Log.e(TAG, "OpenCV initialization failed!");
            return false;
        }

        try {
            // 加载 Haar 人脸分类器
            InputStream is = context.getResources().openRawResource(
                    context.getResources().getIdentifier(
                            "haarcascade_frontalface_alt2", "raw", context.getPackageName()));
            File cascadeDir = context.getDir("cascade", android.content.Context.MODE_PRIVATE);
            File cascadeFile = new File(cascadeDir, "haarcascade_frontalface_alt2.xml");
            FileOutputStream os = new FileOutputStream(cascadeFile);

            byte[] buffer = new byte[4096];
            int bytesRead;
            while ((bytesRead = is.read(buffer)) != -1) {
                os.write(buffer, 0, bytesRead);
            }
            is.close();
            os.close();

            faceClassifier = new CascadeClassifier(cascadeFile.getAbsolutePath());
            if (faceClassifier.empty()) {
                Log.e(TAG, "Failed to load cascade classifier");
                faceClassifier = null;
                return false;
            }

            Log.i(TAG, "Loaded cascade classifier from " + cascadeFile.getAbsolutePath());
            cascadeDir.delete();
            isOpenCVLoaded = true;
            return true;

        } catch (Exception e) {
            Log.e(TAG, "Error loading cascade: " + e.getMessage());
            return false;
        }
    }

    public void setProcessMode(int mode) {
        this.currentProcessMode = mode;
    }

    public int getProcessMode() {
        return currentProcessMode;
    }

    /*
在预览帧上进行人脸检测。
返回检测到的人脸矩形列表（已缩放到目标 View 尺寸）。
     */
    public List<RectF> detectFaces(Bitmap frameBitmap, int targetWidth, int targetHeight) {
        List<RectF> results = new ArrayList<>();
        if (!isOpenCVLoaded || faceClassifier == null || frameBitmap == null) {
            return results;
        }

        Mat rgba = new Mat();
        Mat gray = new Mat();

        try {
            Utils.bitmapToMat(frameBitmap, rgba);
            Imgproc.cvtColor(rgba, gray, Imgproc.COLOR_RGBA2GRAY);

            // 降采样加速检测
            Mat smallGray = new Mat();
            float scaleFactor = 0.5f;
            Imgproc.resize(gray, smallGray, new Size(), scaleFactor, scaleFactor, Imgproc.INTER_LINEAR);

            MatOfRect faces = new MatOfRect();
            faceClassifier.detectMultiScale(
                    smallGray,
                    faces,
                    1.15,     // scaleFactor
                    5,        // minNeighbors（越大越精确，越少误检）
                    0,        // flags
                    new Size(30 * scaleFactor, 30 * scaleFactor),  // minSize
                    new Size()  // maxSize
            );

            Rect[] facesArray = faces.toArray();
            for (Rect r : facesArray) {
                // 转换回原始尺寸坐标，再缩放到目标 View 坐标
                float left = (r.x / scaleFactor) * targetWidth / frameBitmap.getWidth();
                float top = (r.y / scaleFactor) * targetHeight / frameBitmap.getHeight();
                float right = ((r.x + r.width) / scaleFactor) * targetWidth / frameBitmap.getWidth();
                float bottom = ((r.y + r.height) / scaleFactor) * targetHeight / frameBitmap.getHeight();
                results.add(new RectF(left, top, right, bottom));
            }

            smallGray.release();
            faces.release();
        } catch (Exception e) {
            Log.e(TAG, "Face detection error: " + e.getMessage());
        } finally {
            rgba.release();
            gray.release();
        }

        return results;
    }

    /**
拍照后处理。根据当前模式应用不同的图像处理算法。
     /
    public Bitmap processPhoto(Bitmap input) {
        if (!isOpenCVLoaded || input == null) return input;

        Mat src = new Mat();
        Utils.bitmapToMat(input, src);
        Mat result = new Mat();

        try {
            switch (currentProcessMode) {
                case MODE_HDR:
                    result = applyHDR(src);
                    break;
                case MODE_NIGHT:
                    result = applyNightMode(src);
                    break;
                case MODE_PORTRAIT:
                    result = applyPortraitMode(src);
                    break;
                default:
                    result = applyNormalEnhancement(src);
                    break;
            }

            Bitmap output = Bitmap.createBitmap(input.getWidth(), input.getHeight(),
                    Bitmap.Config.ARGB_8888);
            Utils.matToBitmap(result, output);
            return output;

        } catch (Exception e) {
            Log.e(TAG, "Photo processing error: " + e.getMessage());
            return input;
        } finally {
            src.release();
            result.release();
        }
    }

    /*
常规增强：轻微锐化 + 自动对比度。
     /
    private Mat applyNormalEnhancement(Mat src) {
        Mat result = new Mat();

        // 1. Unsharp Mask 锐化
        Mat blurred = new Mat();
        Imgproc.GaussianBlur(src, blurred, new Size(0, 0), 3);
        Core.addWeighted(src, 1.5, blurred, -0.5, 0, result);
        blurred.release();

        // 2. 轻微的饱和度提升
        Mat hsv = new Mat();
        Imgproc.cvtColor(result, hsv, Imgproc.COLOR_RGB2HSV);
        List<Mat> channels = new ArrayList<>();
        Core.split(hsv, channels);

        // 增加 S 通道
        Mat sChannel = channels.get(1);
        Core.convertScaleAbs(sChannel, sChannel, 1.1, 10);

        Core.merge(channels, hsv);
        Imgproc.cvtColor(hsv, result, Imgproc.COLOR_HSV2RGB);
        hsv.release();

        return result;
    }

    /*
HDR 效果：多曝光合成模拟。
使用 Tone Mapping 算法实现 HDR 效果。
     /
    private Mat applyHDR(Mat src) {
        Mat result = new Mat();

        // 转换为浮点
        Mat floatSrc = new Mat();
        src.convertTo(floatSrc, CvType.CV_32F, 1.0 / 255.0);

        // 模拟多曝光：创建过曝和欠曝版本
        Mat overExposed = new Mat();
        Mat underExposed = new Mat();
        Core.convertScaleAbs(src, overExposed, 1.5, 0);
        Core.convertScaleAbs(src, underExposed, 0.7, 0);

        Mat overFloat = new Mat();
        Mat underFloat = new Mat();
        overExposed.convertTo(overFloat, CvType.CV_32F, 1.0 / 255.0);
        underExposed.convertTo(underFloat, CvType.CV_32F, 1.0 / 255.0);

        // 加权融合
        Mat merged = new Mat();
        Core.addWeighted(floatSrc, 0.4, overFloat, 0.3, 0, merged);
        Mat temp = new Mat();
        Core.addWeighted(merged, 0.7, underFloat, 0.3, 0, temp);

        // Tone Mapping (Reinhard)
        Mat logLuminance = new Mat();
        Mat gray = new Mat();
        Imgproc.cvtColor(temp, gray, Imgproc.COLOR_RGB2GRAY);
        Core.addS(gray, new Scalar(0.01), gray); // 防止 log(0)
        Core.log(gray, logLuminance);

        Scalar meanVal = Core.mean(logLuminance);
        double logMean = meanVal.val[0];
        double key = 0.18;

        // 应用 Reinhard Tone Mapping
        Mat toneMapped = new Mat();
        for (int c = 0; c < 3; c++) {
            List<Mat> channels = new ArrayList<>();
            Core.split(temp, channels);
            Mat channel = channels.get(c);

            Mat scaled = new Mat();
            Core.multiply(channel, new Scalar(key / Math.exp(logMean)), scaled);

            Mat mapped = new Mat();
            Core.addS(scaled, new Scalar(1.0), mapped);
            Mat divided = new Mat();
            Core.divide(scaled, mapped, divided);

            // 限制在 [0, 1] 范围
            Core.min(divided, new Scalar(1.0), divided);

            channels.set(c, divided);
            Core.merge(channels, toneMapped);

            scaled.release();
            mapped.release();
            divided.release();
            for (Mat ch : channels) ch.release();
        }

        // 转回 8-bit
        toneMapped.convertTo(result, CvType.CV_8U, 255.0);

        // 锐化
        Mat blurred = new Mat();
        Imgproc.GaussianBlur(result, blurred, new Size(0, 0), 2);
        Core.addWeighted(result, 1.3, blurred, -0.3, 0, result);

        // 释放
        floatSrc.release();
        overExposed.release();
        underExposed.release();
        overFloat.release();
        underFloat.release();
        merged.release();
        temp.release();
        logLuminance.release();
        gray.release();
        toneMapped.release();
        blurred.release();

        return result;
    }

    /*
夜景模式：CLAHE + 降噪 + 亮度提升。
     /
    private Mat applyNightMode(Mat src) {
        Mat result = src.clone();

        // 1. 转 LAB 色彩空间
        Mat lab = new Mat();
        Imgproc.cvtColor(result, lab, Imgproc.COLOR_RGB2Lab);

        List<Mat> labChannels = new ArrayList<>();
        Core.split(lab, labChannels);

        // 2. 对 L 通道应用 CLAHE（自适应直方图均衡化）
        Mat lChannel = labChannels.get(0);
        CLAHE clahe = Imgproc.createCLAHE(3.0, new org.opencv.core.Size(8, 8));
        Mat claheResult = new Mat();
        clahe.apply(lChannel, claheResult);
        labChannels.set(0, claheResult);

        // 3. 轻微降低 b 通道（减少黄色色偏）
        Mat bChannel = labChannels.get(2);
        Core.subtract(bChannel, new Scalar(5), bChannel);

        Core.merge(labChannels, lab);
        Imgproc.cvtColor(lab, result, Imgproc.COLOR_Lab2RGB);

        // 4. 双边滤波降噪（保边）
        Mat denoised = new Mat();
        Imgproc.bilateralFilter(result, denoised, 9, 75, 75);

        // 5. 轻微锐化恢复细节
        Mat blurred = new Mat();
        Imgproc.GaussianBlur(denoised, blurred, new Size(0, 0), 1.5);
        Core.addWeighted(denoised, 1.3, blurred, -0.3, 0, result);

        // 释放
        lab.release();
        for (Mat ch : labChannels) ch.release();
        claheResult.release();
        denoised.release();
        blurred.release();

        return result;
    }

    /*
人像模式：中心清晰 + 边缘虚化（模拟景深）。
     */
    private Mat applyPortraitMode(Mat src) {
        Mat result = new Mat();

        // 1. 对整张图做高斯模糊
        Mat blurred = new Mat();
        Imgproc.GaussianBlur(src, blurred, new Size(21, 21), 0);

        // 2. 创建中心权重蒙版（椭圆形渐变）
        Mat mask = new Mat(src.size(), CvType.CV_32FC1, new Scalar(0));
        int cx = src.cols() / 2;
        int cy = src.rows() / 2;
        int rx = (int) (src.cols() * 0.3);  // 中心清晰区半径
        int ry = (int) (src.rows() * 0.35);

        for (int y = 0; y < src.rows(); y++) {
            for (int x = 0; x < src.cols(); x++) {
                double dx = (double) (x - cx) / rx;
                double dy = (double) (y - cy) / ry;
                double dist = Math.sqrt(dx * dx + dy * dy);
                double weight = Math.min(1.0, Math.max(0.0, (dist - 0.5) / 0.8));
                mask.put(y, x, weight);
            }
        }

        // 3. 混合：中心用原图，边缘用模糊图
        Mat mask3 = new Mat();
        List<Mat> maskChannels = new ArrayList<>();
        maskChannels.add(mask);
        maskChannels.add(mask);
        maskChannels.add(mask);
        Core.merge(maskChannels, mask3);

        Mat invMask3 = new Mat();
        Core.subtract(new Scalar(1.0, 1.0, 1.0), mask3, invMask3);

        Mat srcFloat = new Mat();
        Mat blurFloat = new Mat();
        src.convertTo(srcFloat, CvType.CV_32F, 1.0 / 255.0);
        blurred.convertTo(blurFloat, CvType.CV_32F, 1.0 / 255.0);

        Mat srcMasked = new Mat();
        Mat blurMasked = new Mat();
        Core.multiply(srcFloat, invMask3, srcMasked);
        Core.multiply(blurFloat, mask3, blurMasked);

        Mat blended = new Mat();
        Core.add(srcMasked, blurMasked, blended);

        blended.convertTo(result, CvType.CV_8U, 255.0);

        // 4. 对中心区域轻微增强
        Mat centerEnhanced = new Mat();
        Core.addWeighted(result, 1.05, new Mat(result.size(), result.type(), new Scalar(0)), 0, -5, centerEnhanced);
        result = centerEnhanced;

        // 释放
        blurred.release();
        mask.release();
        mask3.release();
        invMask3.release();
        srcFloat.release();
        blurFloat.release();
        srcMasked.release();
        blurMasked.release();
        blended.release();
        for (Mat ch : maskChannels) ch.release();

        return result;
    }

    /**
释放资源。
     /
    public void release() {
        if (faceClassifier != null) {
            faceClassifier = null;
        }
        isOpenCVLoaded = false;
    }

    public boolean isOpenCVLoaded() {
        return isOpenCVLoaded;
    }

    private void quitThread(HandlerThread thread) {

            thread.quitSafely();
            try {
                thread.join(2000);
            } catch (InterruptedException e) {

        }

    /*
获取相机参数（预览尺寸、拍照尺寸等）。
必须在 openCamera 之前调用。
     /
    public void setUpCameraOutputs(int textureWidth, int textureHeight) {
        try {
            for (String id : cameraManager.getCameraIdList()) {
                CameraCharacteristics characteristics = cameraManager.getCameraCharacteristics(id);
                Integer facing = characteristics.get(CameraCharacteristics.LENS_FACING);

                if (facing != null && facing == (isFrontCamera ?
                        CameraCharacteristics.LENS_FACING_FRONT :
                        CameraCharacteristics.LENS_FACING_BACK)) {
                    cameraId = id;
                    cameraCharacteristics = characteristics;
                    break;
                }
            }

            if (cameraId == null) {
                // 兜底：使用第一个相机
                cameraId = cameraManager.getCameraIdList()[0];
                cameraCharacteristics = cameraManager.getCameraCharacteristics(cameraId);
            }

            // 获取传感器方向
            sensorOrientation = cameraCharacteristics.get(CameraCharacteristics.SENSOR_ORIENTATION);

            // 获取人脸检测支持
            int[] faceDetectModes = cameraCharacteristics.get(
                    CameraCharacteristics.STATISTICS_INFO_AVAILABLE_FACE_DETECT_MODES);
            maxFaceDetectMode = 0;
            if (faceDetectModes != null) {
                for (int mode : faceDetectModes) {
                    maxFaceDetectMode = Math.max(maxFaceDetectMode, mode);
                }
            }

            StreamConfigurationMap map = cameraCharacteristics.get(
                    CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP);
            if (map == null) {
                throw new RuntimeException("Cannot get available preview/video sizes");
            }

            // 选择拍照尺寸（最大）
            Size[] outputSizes = map.getOutputSizes(ImageFormat.JPEG);
            captureSize = CameraUtils.chooseLargestPictureSize(outputSizes);

            // 选择预览尺寸（匹配拍照比例，适配 TextureView）
            Size[] previewSizes = map.getOutputSizes(SurfaceTexture.class);
            previewSize = CameraUtils.chooseOptimalPreviewSize(
                    previewSizes,
                    textureWidth, textureHeight,
                    captureSize.getWidth(), captureSize.getHeight(),
                    captureSize);

            Log.i(TAG, "Preview size: " + previewSize + ", Capture size: " + captureSize);

        } catch (CameraAccessException e) {
            Log.e(TAG, "setUpCameraOutputs: " + e.getMessage());
            if (callback != null) callback.onCameraError("无法获取相机参数");
        }
    }

    // ==================== 打开/关闭相机 ====================

    /*
打开相机。使用 Semaphore 防止并发打开。
     /
    public void openCamera(int textureWidth, int textureHeight) {
        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA)
                != PackageManager.PERMISSION_GRANTED) {
            if (callback != null) callback.onCameraError("相机权限未授予");
            return;
        }

        setUpCameraOutputs(textureWidth, textureHeight);

        try {
            // 关键修复：使用 Semaphore 防止竞态条件
            if (!cameraOpenCloseLock.tryAcquire(2500, TimeUnit.MILLISECONDS)) {
                throw new RuntimeException("Timeout waiting to lock camera opening.");
            }

            cameraManager.openCamera(cameraId, cameraStateCallback, backgroundHandler);

        } catch (CameraAccessException e) {
            cameraOpenCloseLock.release();
            Log.e(TAG, "openCamera CameraAccessException: " + e.getMessage());
            if (callback != null) callback.onCameraError("相机访问异常: " + e.getMessage());
        } catch (InterruptedException e) {
            cameraOpenCloseLock.release();
            throw new RuntimeException("Interrupted while trying to lock camera opening.", e);
        }
    }

    /*
关闭相机并释放所有资源。
     /
    public void closeCamera() {
        try {
            cameraOpenCloseLock.acquire();

            // 按顺序关闭：先停 session，再关 device
            if (captureSession != null) {
                captureSession.close();
                captureSession = null;
            }
            if (cameraDevice != null) {
                cameraDevice.close();
                cameraDevice = null;
            }
            if (imageReader != null) {
                imageReader.close();
                imageReader = null;
            }

            isSessionReady = false;
            currentState = STATE_IDLE;

            if (callback != null) callback.onCameraClosed();

        } catch (InterruptedException e) {
            throw new RuntimeException("Interrupted while trying to lock camera closing.", e);
        } finally {
            cameraOpenCloseLock.release();
        }
    }

    /*
创建预览 Session。
     /
    public void createCameraPreviewSession(SurfaceTexture surfaceTexture) {
        try {
            surfaceTexture.setDefaultBufferSize(previewSize.getWidth(), previewSize.getHeight());
            Surface surface = new Surface(surfaceTexture);

            // 创建 ImageReader 用于拍照
            imageReader = ImageReader.newInstance(
                    captureSize.getWidth(), captureSize.getHeight(),
                    ImageFormat.JPEG, 2);  // maxImages=2 防止内存溢出

            imageReader.setOnImageAvailableListener(imageAvailableListener, imageProcessingHandler);

            // 构建预览请求
            previewRequestBuilder = cameraDevice.createCaptureRequest(CameraDevice.TEMPLATE_PREVIEW);
            previewRequestBuilder.addTarget(surface);

            // 设置自动对焦
            previewRequestBuilder.set(CaptureRequest.CONTROL_AF_MODE,
                    CaptureRequest.CONTROL_AF_MODE_CONTINUOUS_PICTURE);

            // 设置闪光灯
            updateFlashMode();

            // 启用人脸检测
            if (faceDetectEnabled && maxFaceDetectMode > 0) {
                previewRequestBuilder.set(CaptureRequest.STATISTICS_FACE_DETECT_MODE,
                        maxFaceDetectMode);
            }

            // 设置自动曝光
            previewRequestBuilder.set(CaptureRequest.CONTROL_AE_MODE,
                    CaptureRequest.CONTROL_AE_MODE_ON);

            previewRequest = previewRequestBuilder.build();

            // 创建 CaptureSession
            List<Surface> outputs = Arrays.asList(surface, imageReader.getSurface());
            cameraDevice.createCaptureSession(outputs, sessionStateCallback, backgroundHandler);

        } catch (CameraAccessException e) {
            Log.e(TAG, "createCameraPreviewSession: " + e.getMessage());
            if (callback != null) callback.onCameraError("预览创建失败: " + e.getMessage());
        }
    }

    /*
执行实际的拍照操作。
     /
    private void captureStillPicture() {
        try {
            if (cameraDevice == null || imageReader == null) return;

            currentState = STATE_PICTURE_TAKEN;

            CaptureRequest.Builder captureBuilder =
                    cameraDevice.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE);
            captureBuilder.addTarget(imageReader.getSurface());

            // 设置自动对焦
            captureBuilder.set(CaptureRequest.CONTROL_AF_MODE,
                    CaptureRequest.CONTROL_AF_MODE_CONTINUOUS_PICTURE);

            // 设置闪光灯
            if (flashEnabled) {
                captureBuilder.set(CaptureRequest.CONTROL_AE_MODE,
                        CaptureRequest.CONTROL_AE_MODE_ON_ALWAYS_FLASH);
            } else {
                captureBuilder.set(CaptureRequest.CONTROL_AE_MODE,
                        CaptureRequest.CONTROL_AE_MODE_ON);
                captureBuilder.set(CaptureRequest.FLASH_MODE, CaptureRequest.FLASH_MODE_OFF);
            }

            // HDR 场景：使用多帧合成模式
            if (hdrEnabled) {
                captureBuilder.set(CaptureRequest.CONTROL_SCENE_MODE,
                        CameraMetadata.CONTROL_SCENE_MODE_HDR);
            }

            // 设置 JPEG 方向
            int rotation = context.getResources().getConfiguration().orientation;
            int deviceOrientation = rotation == android.content.res.Configuration.ORIENTATION_LANDSCAPE ? 90 : 0;
            int jpegOrientation = CameraUtils.getJpegOrientation(cameraCharacteristics, deviceOrientation);
            captureBuilder.set(CaptureRequest.JPEG_ORIENTATION, jpegOrientation);

            // 停止预览 → 拍照 → 恢复预览
            captureSession.stopRepeating();
            captureSession.abortCaptures();

            captureSession.capture(captureBuilder.build(), new CameraCaptureSession.CaptureCallback() {
                @Override
                public void onCaptureCompleted(@NonNull CameraCaptureSession session,
                                                @NonNull CaptureRequest request,
                                                @NonNull TotalCaptureResult result) {
                    Log.i(TAG, "Capture completed");
                    // 解锁 AF 并恢复预览
                    unlockFocus();
                }
            }, backgroundHandler);

        } catch (CameraAccessException e) {
            Log.e(TAG, "captureStillPicture: " + e.getMessage());
            if (callback != null) callback.onCameraError("拍照失败: " + e.getMessage());
        }
    }

    /*
触摸对焦。

    public void focusOnPoint(float x, float y, int previewWidth, int previewHeight) {
        if (cameraCharacteristics == null || previewRequestBuilder == null) return;

        try {
            // 获取传感器区域
            android.graphics.Rect sensorRect = cameraCharacteristics.get(
                    CameraCharacteristics.SENSOR_INFO_ACTIVE_ARRAY_SIZE);
            if (sensorRect == null) return;

            // 将触摸坐标转换为传感器坐标
            int sensorX = (int) (x / previewWidth * sensorRect.width());
            int sensorY = (int) (y / previewHeight * sensorRect.height());

            // 创建对焦区域
            int halfSize = Math.min(sensorRect.width(), sensorRect.height()) / 8;
            int left = Math.max(0, sensorX - halfSize);
            int top = Math.max(0, sensorY - halfSize);
            int right = Math.min(sensorRect.width(), sensorX + halfSize);
            int bottom = Math.min(sensorRect.height(), sensorY + halfSize);

            android.hardware.camera2.params.MeteringRectangle focusArea =
                    new android.hardware.camera2.params.MeteringRectangle(
                            left, top, right - left, bottom - top,
                            android.hardware.camera2.params.MeteringRectangle.METERING_WEIGHT_MAX);

            previewRequestBuilder.set(CaptureRequest.CONTROL_AF_REGIONS,
                    new android.hardware.camera2.params.MeteringRectangle[]{focusArea});
            previewRequestBuilder.set(CaptureRequest.CONTROL_AE_REGIONS,
                    new android.hardware.camera2.params.MeteringRectangle[]{focusArea});
            previewRequestBuilder.set(CaptureRequest.CONTROL_AF_MODE,
                    CaptureRequest.CONTROL_AF_MODE_AUTO);
            previewRequestBuilder.set(CaptureRequest.CONTROL_AF_TRIGGER,
                    CameraMetadata.CONTROL_AF_TRIGGER_START);

            if (captureSession != null) {
                captureSession.capture(previewRequestBuilder.build(), null, backgroundHandler);

                // 恢复连续对焦
                previewRequestBuilder.set(CaptureRequest.CONTROL_AF_TRIGGER,
                        CameraMetadata.CONTROL_AF_TRIGGER_IDLE);
                previewRequest = previewRequestBuilder.build();
                captureSession.setRepeatingRequest(previewRequest, captureCallback, backgroundHandler);
            }
        } catch (CameraAccessException e) {
            Log.e(TAG, "focusOnPoint: " + e.getMessage());
        }
    }

    private final CameraCaptureSession.CaptureCallback captureCallback =
            new CameraCaptureSession.CaptureCallback() {

        private void processResult(CaptureResult result) {
            // 人脸检测回调
            if (faceDetectEnabled) {
                Face[] faces = result.get(CaptureResult.STATISTICS_FACES);
                if (faces != null && callback != null) {
                    callback.onFacesDetected(faces);
                }
            }

            // AF 状态处理
            Integer afState = result.get(CaptureResult.CONTROL_AF_STATE);
            if (afState == null) return;

            switch (currentState) {
                case STATE_PREVIEW:
                    // 正常预览，检查 AF 是否完成（用于触摸对焦反馈）
                    if (afState == CaptureResult.CONTROL_AF_STATE_FOCUSED_LOCKED |
                            afState == CaptureResult.CONTROL_AF_STATE_NOT_FOCUSED_LOCKED) {
                        boolean success = afState == CaptureResult.CONTROL_AF_STATE_FOCUSED_LOCKED;
                        if (callback != null) callback.onAutoFocusComplete(success);
                        // 恢复连续对焦
                        previewRequestBuilder.set(CaptureRequest.CONTROL_AF_MODE,
                                CaptureRequest.CONTROL_AF_MODE_CONTINUOUS_PICTURE);
                    }
                    break;

                case STATE_WAITING_LOCK:
                    if (afState == CaptureResult.CONTROL_AF_STATE_FOCUSED_LOCKED |
                            afState == CaptureResult.CONTROL_AF_STATE_NOT_FOCUSED_LOCKED |
                            afState == CaptureResult.CONTROL_AF_STATE_PASSIVE_FOCUSED |
                            afState == CaptureResult.CONTROL_AF_STATE_PASSIVE_UNFOCUSED) {
                        // 对焦完成，检查是否需要预闪
                        Integer aeState = result.get(CaptureResult.CONTROL_AE_STATE);
                        if (aeState == null || aeState == CaptureResult.CONTROL_AE_STATE_CONVERGED) {
                            currentState = STATE_PICTURE_TAKEN;
                            captureStillPicture();
                        } else {
                            runPreCaptureSequence();
                        }
                    }
                    break;

                case STATE_WAITING_PRECAPTURE:
                    if (afState == CaptureResult.CONTROL_AE_STATE_PRECAPTURE |
                            afState == CaptureResult.CONTROL_AE_STATE_FLASH_REQUIRED) {
                        currentState = STATE_WAITING_NON_PRECAPTURE;
                    }
                    break;

                case STATE_WAITING_NON_PRECAPTURE:
                    if (afState != CaptureResult.CONTROL_AE_STATE_PRECAPTURE) {
                        currentState = STATE_PICTURE_TAKEN;
                        captureStillPicture();
                    }
                    break;
            }
        }

        @Override
        public void onCaptureProgressed(@NonNull CameraCaptureSession session,
                                         @NonNull CaptureRequest request,
                                         @NonNull CaptureResult partialResult) {
            processResult(partialResult);
        }

        @Override
        public void onCaptureCompleted(@NonNull CameraCaptureSession session,
                                        @NonNull CaptureRequest request,
                                        @NonNull TotalCaptureResult result) {
            processResult(result);
        }
    };

    /**
运行预闪序列（用于闪光灯场景）。
     /
    private void runPreCaptureSequence() {
        try {
            previewRequestBuilder.set(CaptureRequest.CONTROL_AE_PRECAPTURE_TRIGGER,
                    CameraMetadata.CONTROL_AE_PRECAPTURE_TRIGGER_START);
            currentState = STATE_WAITING_PRECAPTURE;
            captureSession.capture(previewRequestBuilder.build(), captureCallback, backgroundHandler);
        } catch (CameraAccessException e) {
            Log.e(TAG, "runPreCaptureSequence: " + e.getMessage());
            captureStillPicture();
        }
    }

    /*
ImageReader 回调：获取拍照结果。
     /
    private final ImageReader.OnImageAvailableListener imageAvailableListener =
            new ImageReader.OnImageAvailableListener() {
        @Override
        public void onImageAvailable(ImageReader reader) {
            Image image = reader.acquireLatestImage();
            if (image != null && callback != null) {
                callback.onImageCaptured(image);
            }
        }
    };

    @Override
    public void onImageCaptured(Image image) {
        // 在后台线程中处理图片
        new Thread(() -> {
            try {
                // 获取 JPEG 数据
                ByteBuffer buffer = image.getPlanes()[0].getBuffer();
                byte[] bytes = new byte[buffer.remaining()];
                buffer.get(bytes);
                image.close();

                // 解码为 Bitmap
                Bitmap bitmap = BitmapFactory.decodeByteArray(bytes, 0, bytes.length);
                if (bitmap == null) {
                    mainHandler.post(() -> showError("图片解码失败"));
                    return;
                }

                // 应用 OpenCV 后处理
                mainHandler.post(() -> processingIndicator.setVisibility(View.VISIBLE));

                Bitmap processed = imageProcessor.processPhoto(bitmap);

                // 保存到相册
                Uri uri = CameraUtils.saveBitmapToGallery(requireContext(), processed, "OPENCV");
                lastPhotoUri = uri;

                // 更新缩略图
                mainHandler.post(() -> {
                    updateThumbnail(processed);
                    processingIndicator.setVisibility(View.GONE);
                    saveProgress.setVisibility(View.GONE);

                    if (uri != null) {
                        showSuccess("照片已保存");
                    } else {
                        showError("保存失败");
                    }
                });

                // 回收原始 bitmap
                if (processed != bitmap) {
                    bitmap.recycle();
                }

            } catch (Exception e) {
                Log.e(TAG, "Image capture processing error: " + e.getMessage());
                mainHandler.post(() -> {
                    processingIndicator.setVisibility(View.GONE);
                    showError("照片处理失败: " + e.getMessage());
                });
                image.close();
            }
        }).start();
    }

            cameraController.takePicture();

        }, 100);

    private void startCountdown(int seconds) {
        countdownText.setVisibility(View.VISIBLE);
        countdownText.setText(String.valueOf(seconds));

        ObjectAnimator scaleAnim = ObjectAnimator.ofFloat(countdownText, "scaleX", 1.5f, 1f);
        scaleAnim.setDuration(800);
        scaleAnim.start();
        ObjectAnimator alphaAnim = ObjectAnimator.ofFloat(countdownText, "alpha", 1f, 0.3f);
        alphaAnim.setDuration(1000);
        alphaAnim.start();

        if (seconds > 1) {
            mainHandler.postDelayed(() -> startCountdown(seconds - 1), 1000);
        } else {
            mainHandler.postDelayed(() -> {
                countdownText.setVisibility(View.GONE);
                doCapture();
            }, 800);
        }
    }

    private void updateThumbnail(Bitmap bitmap) {
        if (bitmap == null || thumbnail == null) return;

        // 裁剪为正方形缩略图
        int size = Math.min(bitmap.getWidth(), bitmap.getHeight());
        int x = (bitmap.getWidth() - size) / 2;
        int y = (bitmap.getHeight() - size) / 2;
        Bitmap cropped = Bitmap.createBitmap(bitmap, x, y, size, size);

        // 缩放为缩略图尺寸
        Bitmap thumbBitmap = Bitmap.createScaledBitmap(cropped, 120, 120, true);

        thumbnail.setImageBitmap(thumbBitmap);

        if (cropped != bitmap) cropped.recycle();

        // 缩略图出现动画
        thumbnail.animate()
                .scaleX(0.8f).scaleY(0.8f)
                .setDuration(100)
                .withEndAction(() -> thumbnail.animate()
                        .scaleX(1f).scaleY(1f)
                        .setDuration(200)
                        .setInterpolator(new OvershootInterpolator(2f))
                        .start())
                .start();
    }

    private void setupGestureDetectors() {

                    @Override
                    public boolean onSingleTapUp(MotionEvent e) {

                        cameraController.focusOnPoint(
                                e.getX(), e.getY(),
                                textureView.getWidth(), textureView.getHeight());
                        return true;
                    }
                });

        ScaleGestureDetector scaleDetector = new ScaleGestureDetector(requireContext(),
                new ScaleGestureDetector.SimpleOnScaleGestureListener() {
                    @Override
                    public boolean onScale(ScaleGestureDetector detector) {
                        currentZoom = detector.getScaleFactor();
                        currentZoom = Math.max(1f, Math.min(currentZoom, 5f));
                        // TODO: 应用数字变焦到 CameraController
                        return true;
                    }
                });

        textureView.setOnTouchListener((v, event) -> {
            gestureDetector.onTouchEvent(event);
            scaleDetector.onTouchEvent(event);
            return true;
        });
    }

    // ==================== 人脸检测（OpenCV 软件检测） ====================

    private void onPreviewFrameUpdate(SurfaceTexture surfaceTexture) {
        long now = System.currentTimeMillis();
        if (now - lastFaceDetectTime < FACE_DETECT_INTERVAL) return;
        if (!imageProcessor.isOpenCVLoaded()) return;
        if (cameraController.isFrontCamera()) return; // 前置摄像头用硬件检测

        lastFaceDetectTime = now;

        // 在后台线程中检测（不阻塞 UI）
        new Thread(() -> {
            try {
                // 从 TextureView 获取当前帧
                Bitmap frame = textureView.getBitmap();
                if (frame == null) return;

                // 缩小用于检测
                Bitmap smallFrame = Bitmap.createScaledBitmap(frame, frame.getWidth() / 4,
                        frame.getHeight() / 4, true);

                List<android.graphics.RectF> faces = imageProcessor.detectFaces(
                        smallFrame, faceOverlay.getWidth(), faceOverlay.getHeight());

                if (!faces.isEmpty()) {
                    mainHandler.post(() -> faceOverlay.updateOpenCVFaces(faces));
                } else {
                    mainHandler.post(faceOverlay::clearFaces);
                }

                smallFrame.recycle();
                // 不回收 frame，因为 getBitmap() 每次创建新的
            } catch (Exception e) {
                // 静默处理
            }
        }).start();
    }

    private void configureTransform(int viewWidth, int viewHeight) {

                viewWidth, viewHeight,
                previewSize.getWidth(), previewSize.getHeight(),

        textureView.setTransform(matrix);

// 全局人脸检测器（避免重复加载）
static CascadeClassifier g_faceCascade = nullptr;

extern "C" {

/**
初始化人脸检测器。
 */
JNIEXPORT jboolean JNICALL
Java_com_opencv_camera_ImageProcessor_nativeInitFaceDetector(
        JNIEnv env, jobject / this */, jstring cascadePath) {

    const char path = env->GetStringUTFChars(cascadePath, nullptr);
    if (g_faceCascade == nullptr) {
        g_faceCascade = new CascadeClassifier();
    }

    bool loaded = g_faceCascade->load(path);
    env->ReleaseStringUTFChars(cascadePath, path);

    LOGI("Face cascade loaded: %s", loaded ? "true" : "false");
    return loaded ? JNI_TRUE : JNI_FALSE;

/*
高性能人脸检测 - 直接在 Native 层操作 Mat。
避免 Java ↔ Native 的 Bitmap 数据拷贝。
* @param matAddr  源图像 Mat 的 nativeObj 地址
@param result   输出浮点数组 [x1, y1, w1, h1, x2, y2, w2, h2, ...]
@return 检测到的人脸数量
 */
JNIEXPORT jint JNICALL
Java_com_opencv_camera_ImageProcessor_nativeDetectFaces(
        JNIEnv env, jobject / this */, jlong matAddr, jfloatArray result) {

    if (g_faceCascade == nullptr || matAddr == 0) return 0;

    Mat &src = *(Mat ) matAddr;
    Mat gray;

    // 转灰度
    if (src.channels() == 4) {
        cvtColor(src, gray, COLOR_RGBA2GRAY);
    } else if (src.channels() == 3) {
        cvtColor(src, gray, COLOR_RGB2GRAY);
    } else {
        gray = src.clone();
    }

    // 降采样加速（缩小到 50%）
    Mat small;
    float scaleFactor = 0.5f;
    resize(gray, small, Size(), scaleFactor, scaleFactor, INTER_LINEAR);

    // 直方图均衡化（提高检测率）
    equalizeHist(small, small);

    // 执行检测
    vector<Rect> faces;
    g_faceCascade->detectMultiScale(
            small,
            faces,
            1.15,   // scaleFactor
            5,      // minNeighbors (越高越精确)
            0,      // flags
            Size(30, 30),   // minSize
            Size()          // maxSize (无限制)
    );

    // 转换回原始尺寸
    g_detectedFaces.clear();
    for (auto &face : faces) {
        Rect originalRect(
                (int)(face.x / scaleFactor),
                (int)(face.y / scaleFactor),
                (int)(face.width / scaleFactor),
                (int)(face.height / scaleFactor)
        );
        g_detectedFaces.push_back(originalRect);
    }

    // 填充结果数组
    int faceCount = min((int) g_detectedFaces.size(), (int)(env->GetArrayLength(result) / 4));
    if (faceCount > 0) {
        jfloat resultPtr = env->GetFloatArrayElements(result, nullptr);
        for (int i = 0; i < faceCount; i++) {
            resultPtr[i * 4 + 0] = (float) g_detectedFaces[i].x;
            resultPtr[i * 4 + 1] = (float) g_detectedFaces[i].y;
            resultPtr[i * 4 + 2] = (float) g_detectedFaces[i].width;
            resultPtr[i * 4 + 3] = (float) g_detectedFaces[i].height;
        }
        env->ReleaseFloatArrayElements(result, resultPtr, 0);
    }

    LOGD("Detected %d faces", faceCount);
    return faceCount;
}

/**
Unsharp Mask 锐化。
直接在 Bitmap 像素上操作，零拷贝。
* @param bitmap 输入/输出 Bitmap (ARGB_8888)
@param strength 锐化强度 (0.0 ~ 2.0)
@param radius 高斯模糊半径
 */
JNIEXPORT void JNICALL
Java_com_opencv_camera_ImageProcessor_nativeSharpen(
        JNIEnv env, jobject / this /, jobject bitmap,
        jfloat strength, jfloat radius) {

    AndroidBitmapInfo info;
    void pixels = lockBitmapPixels(env, bitmap, &info);
    if (!pixels) return;

    // 将 Bitmap 像素包装为 Mat（零拷贝）
    Mat src(info.height, info.width, CV_8UC4, pixels);

    // Unsharp Mask: result = src + strength * (src - blurred)
    Mat blurred;
    GaussianBlur(src, blurred, Size(0, 0), radius);

    // 混合：addWeighted(src, 1+strength, blurred, -strength, 0)
    addWeighted(src, 1.0 + strength, blurred, -strength, 0, src);

    // 限制像素范围 [0, 255]
    // addWeighted 已经内部处理了饱和截断

    blurred.release();
    unlockBitmapPixels(env, bitmap);

    LOGD("Sharpen applied: strength=%.2f, radius=%.2f", strength, radius);
}

/**
人像模式背景虚化（Bokeh 效果）。
使用深度估计（简化版：基于中心距离）+ 高斯模糊。
* @param bitmap    输入/输出 Bitmap
@param centerX   对焦中心 X (0~1 归一化坐标)
@param centerY   对焦中心 Y
@param focusRadius 对焦半径 (0~1)
@param blurStrength 虚化强度
 */
JNIEXPORT void JNICALL
Java_com_opencv_camera_ImageProcessor_nativeBokeh(
        JNIEnv env, jobject / this /, jobject bitmap,
        jfloat centerX, jfloat centerY,
        jfloat focusRadius, jfloat blurStrength) {

    AndroidBitmapInfo info;
    void pixels = lockBitmapPixels(env, bitmap, &info);
    if (!pixels) return;

    int w = info.width;
    int h = info.height;

    Mat src(h, w, CV_8UC4, pixels);

    // 1. 创建强模糊版本
    Mat blurred;
    int kernelSize = (int)(blurStrength * 2) * 2 + 1; // 确保为奇数
    if (kernelSize < 3) kernelSize = 3;
    GaussianBlur(src, blurred, Size(kernelSize, kernelSize), 0);

    // 多次模糊叠加，模拟更自然的 Bokeh
    for (int i = 0; i < 3; i++) {
        GaussianBlur(blurred, blurred, Size(kernelSize, kernelSize), 0);
    }

    // 2. 创建深度蒙版（椭圆形渐变）
    Mat mask(h, w, CV_32FC1);
    int cx = (int)(centerX * w);
    int cy = (int)(centerY * h);
    float rx = focusRadius * w;
    float ry = focusRadius * h * 1.2f; // 稍微竖向拉长

    for (int y = 0; y < h; y++) {
        float* maskRow = mask.ptr<float>(y);
        for (int x = 0; x < w; x++) {
            float dx = (float)(x - cx) / rx;
            float dy = (float)(y - cy) / ry;
            float dist = sqrtf(dx * dx + dy * dy);
            // 平滑过渡：0（完全清晰）到 1（完全模糊）
            float weight = (dist - 0.6f) / 0.6f;
            weight = max(0.0f, min(1.0f, weight));
            // 使用 smoothstep 让过渡更自然
            weight = weight * weight * (3.0f - 2.0f * weight);
            maskRow[x] = weight;
        }
    }

    // 3. 混合：mask=0 用原图，mask=1 用模糊图
    Mat srcFloat, blurFloat, result;
    src.convertTo(srcFloat, CV_32FC4, 1.0 / 255.0);
    blurred.convertTo(blurFloat, CV_32FC4, 1.0 / 255.0);

    // 扩展蒙版为 4 通道
    vector<Mat> maskChannels = {mask, mask, mask, mask};
    Mat mask4;
    merge(maskChannels, mask4);

    Mat invMask4;
    subtract(Scalar::all(1.0), mask4, invMask4);

    Mat srcMasked, blurMasked;
    multiply(srcFloat, invMask4, srcMasked);
    multiply(blurFloat, mask4, blurMasked);

    add(srcMasked, blurMasked, result);

    // 4. 对清晰区域轻微增强（让主体更突出）
    Mat enhanced;
    result.convertTo(enhanced, CV_8UC4, 255.0);

    // 写回 Bitmap
    enhanced.copyTo(src);

    // 释放
    blurred.release();
    mask.release();
    srcFloat.release();
    blurFloat.release();
    mask4.release();
    invMask4.release();
    srcMasked.release();
    blurMasked.release();
    result.release();
    enhanced.release();

    unlockBitmapPixels(env, bitmap);

    LOGD("Bokeh applied: center=(%.2f,%.2f), radius=%.2f, blur=%.2f",
         centerX, centerY, focusRadius, blurStrength);
}

/**
实时预览滤镜（高性能版本）。
用于预览帧的实时处理，优化了性能。
* @param yuvAddr   YUV 帧的 Mat 地址
@param outAddr   输出帧的 Mat 地址
@param mode      滤镜模式 0=normal, 1=vivid, 2=warm, 3=cool, 4=bw
 */
JNIEXPORT void JNICALL
Java_com_opencv_camera_ImageProcessor_nativePreviewFilter(
        JNIEnv env, jobject / this */,
        jlong yuvAddr, jlong outAddr, jint mode) {

    if (yuvAddr == 0 || outAddr == 0) return;

    Mat &yuv = *(Mat *) yuvAddr;
    Mat &out = *(Mat ) outAddr;

    // YUV → RGBA
    cvtColor(yuv, out, COLOR_YUV2RGBA_NV21);

    switch (mode) {
        case 1: { // Vivid（鲜艳）
            // 提高饱和度和对比度
            Mat hsv;
            cvtColor(out, hsv, COLOR_RGBA2RGB);
            Mat rgb;
            cvtColor(hsv, rgb, COLOR_RGB2HSV);

            vector<Mat> channels;
            split(rgb, channels);

            // 增加饱和度
            channels[1].convertTo(channels[1], -1, 1.3, 10);

            merge(channels, rgb);
            cvtColor(rgb, hsv, COLOR_HSV2RGB);
            cvtColor(hsv, out, COLOR_RGB2RGBA);

            // 增加对比度
            out.convertTo(out, -1, 1.1, -10);

            hsv.release();
            rgb.release();
            break;
        }
        case 2: { // Warm（暖色调）
            vector<Mat> channels;
            split(out, channels);
            // 增加红色，减少蓝色
            channels[0].convertTo(channels[0], -1, 1.0, 15);  // R + 15
            channels[2].convertTo(channels[2], -1, 1.0, -10); // B - 10
            merge(channels, out);
            break;
        }
        case 3: { // Cool（冷色调）
            vector<Mat> channels;
            split(out, channels);
            channels[0].convertTo(channels[0], -1, 1.0, -10); // R - 10
            channels[2].convertTo(channels[2], -1, 1.0, 15);  // B + 15
            merge(channels, out);
            break;
        }
        case 4: { // B&W（黑白）
            Mat gray;
            cvtColor(out, gray, COLOR_RGBA2GRAY);
            cvtColor(gray, out, COLOR_GRAY2RGBA);
            gray.release();
            break;
        }
        default: // Normal - 不做处理
            break;
    }
}

/*
多帧降噪（模拟 Google Camera 的夜景模式）。
对连续多帧进行对齐和平均，大幅降低噪声。
* @param frameAddresses  多帧 Mat 地址数组
@param frameCount      帧数
@param outAddr         输出 Mat 地址
 */
JNIEXPORT void JNICALL
Java_com_opencv_camera_ImageProcessor_nativeMultiFrameNR(
        JNIEnv env, jobject / this /,
        jlongArray frameAddresses, jint frameCount, jlong outAddr) {

    if (frameCount < 2 || outAddr == 0) return;

    jlong addrs = env->GetLongArrayElements(frameAddresses, nullptr);
    Mat &result = *(Mat *) outAddr;

    // 第一帧作为参考帧
    Mat &refFrame = *(Mat *) addrs[0];
    Mat accumulator;
    refFrame.convertTo(accumulator, CV_32F);

    // 对后续帧进行对齐并累加
    for (int i = 1; i < frameCount; i++) {
        Mat &frame = *(Mat ) addrs[i];
        Mat aligned;

        // 使用 ECC（增强相关系数）算法对齐帧
        Mat warpMatrix = Mat::eye(2, 3, CV_32F);
        try {
            Mat refGray, frameGray;
            cvtColor(refFrame, refGray, COLOR_RGBA2GRAY);
            cvtColor(frame, frameGray, COLOR_RGBA2GRAY);

            // 降采样加速对齐
            Mat smallRef, smallFrame;
            resize(refGray, smallRef, Size(), 0.25, 0.25);
            resize(frameGray, smallFrame, Size(), 0.25, 0.25);

            findTransformECC(smallRef, smallFrame, warpMatrix,
                             MOTION_TRANSLATION,
                             TermCriteria(TermCriteria::COUNT + TermCriteria::EPS, 50, 0.001));

            // 应用变换
            warpAffine(frame, aligned, warpMatrix, frame.size(),
                       INTER_LINEAR + WARP_INVERSE_MAP);
        } catch (const cv::Exception &e) {
            LOGE("Frame alignment failed for frame %d: %s", i, e.what());
            aligned = frame.clone();
        }

        // 累加
        Mat frameFloat;
        aligned.convertTo(frameFloat, CV_32F);
        accumulator += frameFloat;

        aligned.release();
    }

    // 平均
    accumulator /= (float) frameCount;

    // 转回 8-bit
    accumulator.convertTo(result, CV_8UC4);

    env->ReleaseLongArrayElements(frameAddresses, addrs, 0);
    accumulator.release();

    LOGI("Multi-frame NR completed: %d frames averaged", frameCount);
}

释放全局资源。

}

GridOverlayView.java

package com.opencv.camera;

import android.content.Context;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.util.AttributeSet;
import android.view.View;

import androidx.annotation.Nullable;

/*
九宫格辅助线覆盖层。
仿 Google Camera 风格：细白线，半透明。
支持三分法和黄金比例两种模式。
 /
public class GridOverlayView extends View {

    public static final int GRID_THIRDS = 0;       // 三分法
    public static final int GRID_GOLDEN = 1;        // 黄金比例
    public static final int GRID_DIAGONAL = 2;      // 对角线

    private final Paint gridPaint;
    private int gridMode = GRID_THIRDS;

    public GridOverlayView(Context context) {
        this(context, null);
    }

    public GridOverlayView(Context context, @Nullable AttributeSet attrs) {
        this(context, attrs, 0);
    }

    public GridOverlayView(Context context, @Nullable AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);

        gridPaint = new Paint(Paint.ANTI_ALIAS_FLAG);
        gridPaint.setStyle(Paint.Style.STROKE);
        gridPaint.setStrokeWidth(1.0f);
        gridPaint.setColor(Color.parseColor("#33FFFFFF")); // 半透明白色
    }

    public void setGridMode(int mode) {
        this.gridMode = mode;
        invalidate();
    }

    public int getGridMode() {
        return gridMode;
    }

    /*
循环切换网格模式。
     /
    public int cycleGridMode() {
        gridMode = (gridMode + 1) % 3;
        invalidate();
        return gridMode;
    }

    @Override
    protected void onDraw(Canvas canvas) {
        super.onDraw(canvas);

        int w = getWidth();
        int h = getHeight();

        if (w == 0 || h == 0) return;

        switch (gridMode) {
            case GRID_THIRDS:
                drawThirdsGrid(canvas, w, h);
                break;
            case GRID_GOLDEN:
                drawGoldenGrid(canvas, w, h);
                break;
            case GRID_DIAGONAL:
                drawDiagonalGrid(canvas, w, h);
                break;
        }
    }

    /*
三分法网格。
     */
    private void drawThirdsGrid(Canvas canvas, int w, int h) {
        float thirdW = w / 3f;
        float thirdH = h / 3f;

        // 竖线
        canvas.drawLine(thirdW, 0, thirdW, h, gridPaint);
        canvas.drawLine(thirdW * 2, 0, thirdW * 2, h, gridPaint);

        // 横线
        canvas.drawLine(0, thirdH, w, thirdH, gridPaint);
        canvas.drawLine(0, thirdH * 2, w, thirdH * 2, gridPaint);
    }

    /**
黄金比例网格 (φ ≈ 1.618)。
     /
    private void drawGoldenGrid(Canvas canvas, int w, int h) {
        float phi = 1.618033988749895f;
        float x1 = w / phi;
        float x2 = w - x1;
        float y1 = h / phi;
        float y2 = h - y1;

        // 稍微改变颜色以示区分
        Paint goldenPaint = new Paint(gridPaint);
        goldenPaint.setColor(Color.parseColor("#44FFD700")); // 金色半透明

        canvas.drawLine(x1, 0, x1, h, goldenPaint);
        canvas.drawLine(x2, 0, x2, h, goldenPaint);
        canvas.drawLine(0, y1, w, y1, goldenPaint);
        canvas.drawLine(0, y2, w, y2, goldenPaint);
    }

    /*
对角线 + 中心十字。
     /
    private void drawDiagonalGrid(Canvas canvas, int w, int h) {
        Paint diagPaint = new Paint(gridPaint);
        diagPaint.setColor(Color.parseColor("#22FFFFFF"));

        // 对角线
        canvas.drawLine(0, 0, w, h, diagPaint);
        canvas.drawLine(w, 0, 0, h, diagPaint);

        // 中心十字（更细更淡）
        Paint centerPaint = new Paint(gridPaint);
        centerPaint.setStrokeWidth(0.5f);
        centerPaint.setColor(Color.parseColor("#44FFFFFF"));

        canvas.drawLine(w / 2f, 0, w / 2f, h, centerPaint);
        canvas.drawLine(0, h / 2f, w, h / 2f, centerPaint);
    }
}

public class SettingsFragment extends Fragment {

    private View createSwitchItem(String title, String subtitle, String key, boolean defaultValue) {
        LinearLayout item = new LinearLayout(requireContext());

        item.setPadding(dp(24), dp(16), dp(24), dp(16));
        item.setGravity(android.view.Gravity.CENTER_VERTICAL);

        titleView.setTextSize(16);

        textContainer.addView(subtitleView);

                0, ViewGroup.LayoutParams.WRAP_CONTENT, 1);
        item.addView(textContainer, textParams);

        // 右侧开关

        switchView.setChecked(prefs.getBoolean(key, defaultValue));

        container.addView(item);

        divider.setBackgroundColor(0x1AFFFFFF);
        container.addView(divider, new LinearLayout.LayoutParams(
                ViewGroup.LayoutParams.MATCH_PARENT, 1));

        return container;
    }

    }
